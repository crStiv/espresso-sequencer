23/// Deploy contracts needed to run the sequencer.
24///
25/// This script deploys contracts needed to run the sequencer to an L1. It outputs a .env file
26/// containing the addresses of the deployed contracts.
27///
28/// This script can also be used to do incremental deployments. The only contract addresses
29/// needed to configure the sequencer network are ESPRESSO_SEQUENCER_FEE_CONTRACT_PROXY_ADDRESS,
30/// ESPRESSO_SEQUENCER_LIGHT_CLIENT_PROXY_ADDRESS and ESPRESSO_SEQUENCER_STAKE_TABLE_ADDRESS.
31/// These contracts, however, have dependencies, and a full deployment involves several
32/// contracts. Some of these contracts, especially libraries may already have been deployed, or
33/// perhaps one of the top-level contracts has been deployed and we only need to deploy the other
34/// one.
35///
36/// It is possible to pass in the addresses of already deployed contracts, in which case those
37/// addresses will be used in place of deploying a new contract wherever that contract is required
38/// in the deployment process. The generated .env file will include all the addresses passed in as
39/// well as those newly deployed.

276#[tokio::main]
277async fn main() -> anyhow::Result<()> {
278    let opt = Options::parse();
279
280    opt.logging.init();
281
282    if matches!(opt.command, Some(Command::VerifyNodeJsFiles)) {
283        verify_node_js_files().await?;
284        return Ok(());
285    };
286
287    let mut contracts = Contracts::from(opt.contracts);
288    let provider = if opt.ledger {
289        let signer = connect_ledger(opt.account_index as usize).await?;
290        tracing::info!("Using ledger for signing, watch ledger device for prompts.");
291        build_provider_ledger(signer, opt.rpc_url.clone(), Some(opt.l1_polling_interval))
292    } else {
293        build_provider(
294            opt.mnemonic
295                .expect("Mnemonic provided when not using ledger"),
296            opt.account_index,
297            opt.rpc_url.clone(),
298            Some(opt.l1_polling_interval),
299        )
300    };
301
302    // Fail early if we can't connect to the Ethereum RPC.
303    let chain_id = provider.get_chain_id().await.with_context(|| {
304        // The URL may contain a key in the query string.
305        let mut url = opt.rpc_url.clone();
306        url.set_query(None);
307        format!("Unable query Ethereum provider {url}..")
308    })?;
309    tracing::info!("Connected to chain with chain ID: {chain_id}");
310
311    let account = provider.default_signer_address();
312    if let Some(command) = &opt.command {
313        match command {
314            Command::Account => {
315                println!("{account}");
316                return Ok(());
317            },
318            Command::Balance => {
319                let balance = provider.get_balance(account).await?;
320                println!("{account}: {} Eth", format_ether(balance));
321                return Ok(());
322            },
323            _ => unreachable!(),
324        };
325    };
326
327    // No subcommand specified. Deploy contracts.
328
329    let balance = provider.get_balance(account).await?;
330    tracing::info!(
331        "Using deployer account {account} with balance: {}",
332        format_ether(balance),
333    );
334    if balance.is_zero() {
335        anyhow::bail!(
336            "account_index {}, address={account} has no balance. A funded account is required.",
337            opt.account_index
338        );
339    }
340
341    // First use builder to build constructor input arguments
342    let mut args_builder = DeployerArgsBuilder::default();
343    args_builder
344        .deployer(provider)
345        .mock_light_client(opt.use_mock)
346        .use_multisig(opt.use_multisig)
347        .dry_run(opt.dry_run)
348        .rpc_url(opt.rpc_url.to_string());
349    if let Some(multisig) = opt.multisig_address {
350        args_builder.multisig(multisig);
351    }
352    if let Some(multisig_pauser) = opt.multisig_pauser_address {
353        args_builder.multisig_pauser(multisig_pauser);
354    }
355    if let Some(token_recipient) = opt.initial_token_grant_recipient {
356        args_builder.token_recipient(token_recipient);
357    }
358
359    if let Some(blocks_per_epoch) = opt.blocks_per_epoch {
360        args_builder.blocks_per_epoch(blocks_per_epoch);
361    }
362    if let Some(epoch_start_block) = opt.epoch_start_block {
363        args_builder.epoch_start_block(epoch_start_block);
364    }
365
366    if opt.deploy_light_client_v1 {
367        let (genesis_state, genesis_stake) = if opt.mock_espresso_live_network {
368            light_client_genesis_from_stake_table(&Default::default(), DEFAULT_STAKE_TABLE_CAPACITY)
369                .unwrap()
370        } else {
371            light_client_genesis(&opt.sequencer_url, opt.stake_table_capacity).await?
372        };
373        args_builder
374            .genesis_lc_state(genesis_state)
375            .genesis_st_state(genesis_stake);
376        if let Some(prover) = opt.permissioned_prover {
377            args_builder.permissioned_prover(prover);
378        }
379    }
380    if opt.upgrade_light_client_v2 {
381        let (blocks_per_epoch, epoch_start_block) =
382            if (opt.dry_run && opt.use_multisig) || opt.mock_espresso_live_network {
383                (10, 22)
384            } else {
385                // fetch epoch length from HotShot config
386                // Request the configuration until it is successful
387                loop {
388                    match surf_disco::Client::<ServerError, StaticVersion<0, 1>>::new(
389                        opt.sequencer_url.clone(),
390                    )
391                    .get::<PublicNetworkConfig>("config/hotshot")
392                    .send()
393                    .await
394                    {
395                        Ok(resp) => {
396                            let config = resp.hotshot_config();
397                            break (config.blocks_per_epoch(), config.epoch_start_block());
398                        },
399                        Err(e) => {
400                            tracing::error!("Failed to fetch the network config: {e}");
401                            sleep(Duration::from_secs(5));
402                        },
403                    }
404                }
405            };
406        args_builder.blocks_per_epoch(blocks_per_epoch);
407        args_builder.epoch_start_block(epoch_start_block);
408    }
409
410    if opt.deploy_stake_table {
411        if let Some(escrow_period) = opt.exit_escrow_period {
412            args_builder.exit_escrow_period(U256::from(escrow_period.as_secs()));
413        }
414    }
415    if opt.deploy_ops_timelock {
416        let ops_timelock_admin = opt
417            .ops_timelock_admin
418            .expect("Must provide --ops-timelock-admin when deploying ops timelock");
419        args_builder.ops_timelock_admin(ops_timelock_admin);
420        let ops_timelock_delay = opt
421            .ops_timelock_delay
422            .expect("Must provide --ops-timelock-delay when deploying ops timelock");
423        args_builder.ops_timelock_delay(U256::from(ops_timelock_delay));
424        let ops_timelock_executors = opt
425            .ops_timelock_executors
426            .expect("Must provide --ops-timelock-executors when deploying ops timelock");
427        args_builder.ops_timelock_executors(ops_timelock_executors.into_iter().collect());
428        let ops_timelock_proposers = opt
429            .ops_timelock_proposers
430            .expect("Must provide --ops-timelock-proposers when deploying ops timelock");
431        args_builder.ops_timelock_proposers(ops_timelock_proposers.into_iter().collect());
432    }
433
434    if opt.deploy_safe_exit_timelock {
435        let safe_exit_timelock_admin = opt
436            .safe_exit_timelock_admin
437            .expect("Must provide --safe-exit-timelock-admin when deploying token timelock");
438        args_builder.safe_exit_timelock_admin(safe_exit_timelock_admin);
439        let safe_exit_timelock_delay = opt
440            .safe_exit_timelock_delay
441            .expect("Must provide --safe-exit-timelock-delay when deploying token timelock");
442        args_builder.safe_exit_timelock_delay(U256::from(safe_exit_timelock_delay));
443        let safe_exit_timelock_executors = opt
444            .safe_exit_timelock_executors
445            .expect("Must provide --safe-exit-timelock-executors when deploying token timelock");
446        args_builder
447            .safe_exit_timelock_executors(safe_exit_timelock_executors.into_iter().collect());
448        let safe_exit_timelock_proposers = opt
449            .safe_exit_timelock_proposers
450            .expect("Must provide --safe-exit-timelock-proposers when deploying token timelock");
451        args_builder
452            .safe_exit_timelock_proposers(safe_exit_timelock_proposers.into_iter().collect());
453    }
454
455    // then deploy specified contracts
456    let args = args_builder.build()?;
457    if opt.deploy_fee {
458        args.deploy(&mut contracts, Contract::FeeContractProxy)
459            .await?;
460    }
461    if opt.deploy_esp_token {
462        args.deploy(&mut contracts, Contract::EspTokenProxy).await?;
463    }
464    if opt.deploy_light_client_v1 {
465        args.deploy(&mut contracts, Contract::LightClientProxy)
466            .await?;
467    }
468    if opt.upgrade_light_client_v2 {
469        args.deploy(&mut contracts, Contract::LightClientV2).await?;
470    }
471    if opt.deploy_stake_table {
472        args.deploy(&mut contracts, Contract::StakeTableProxy)
473            .await?;
474    }
475    if opt.upgrade_stake_table_v2 {
476        args.deploy(&mut contracts, Contract::StakeTableV2).await?;
477    }
478    if opt.deploy_ops_timelock {
479        args.deploy(&mut contracts, Contract::OpsTimelock).await?;
480    }
481    if opt.deploy_safe_exit_timelock {
482        args.deploy(&mut contracts, Contract::SafeExitTimelock)
483            .await?;
484    }
485
486    // finally print out or persist deployed addresses
487    if let Some(out) = &opt.out {
488        let file = File::options()
489            .create(true)
490            .truncate(true)
491            .write(true)
492            .open(out)?;
493        contracts.write(file)?;
494    } else {
495        contracts.write(stdout())?;
496    }
497
498    Ok(())
499}
